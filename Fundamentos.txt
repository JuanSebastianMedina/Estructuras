public class Main {

    public static void bubbleSort(int[] arr) {

        for (int i = 0; i < arr.length - 1; i++) {
            for (int j = 0; j < arr.length - i - 1; j++) {
                if (arr[j] > arr[j + 1]) {

                    int temp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = temp;
                }
            }
        }
    }

    public static void main(String[] args) {

        int[] numeros = {5, 2, 9, 1};

        bubbleSort(numeros);

        for (int n : numeros)
            System.out.print(n + " ");
    }
}

-> merge sort O(n log n) ---
import java.util.Arrays;

public class Main {

    public static void mergeSort(int[] arr) {
        if (arr.length < 2) return;

        int mid = arr.length / 2;
        int[] left = Arrays.copyOfRange(arr, 0, mid);
        int[] right = Arrays.copyOfRange(arr, mid, arr.length);

        mergeSort(left);
        mergeSort(right);

        merge(arr, left, right);
    }

    public static void merge(int[] arr, int[] left, int[] right) {

        int i = 0, j = 0, k = 0;

        while (i < left.length && j < right.length) {
            if (left[i] <= right[j])
                arr[k++] = left[i++];
            else
                arr[k++] = right[j++];
        }

        while (i < left.length)
            arr[k++] = left[i++];

        while (j < right.length)
            arr[k++] = right[j++];
    }

    public static void main(String[] args) {
        int[] arr = {5, 3, 8, 1};
        mergeSort(arr);
        System.out.println(Arrays.toString(arr));
    }
}

-> busqueda binaria O(log n)
Requiere arreglo ordenado

public class Main {

    public static int binarySearch(int[] arr, int target) {

        int left = 0, right = arr.length - 1;

        while (left <= right) {

            int mid = (left + right) / 2;

            if (arr[mid] == target)
                return mid;

            if (arr[mid] < target)
                left = mid + 1;
            else
                right = mid - 1;
        }

        return -1;
    }

    public static void main(String[] args) {

        int[] arr = {1, 3, 5, 7, 9};
        System.out.println(binarySearch(arr, 7));
    }
}

public static Node insert(Node root, int value) {

    if (root == null)
        return new Node(value);

    if (value < root.value)
        root.left = insert(root.left, value);
    else
        root.right = insert(root.right, value);

    return root;
}

import java.util.*;

public class Main {

    public static void permutar(String str, String resultado) {

        if (str.length() == 0) {
            System.out.println(resultado);
            return;
        }

        for (int i = 0; i < str.length(); i++) {

            char actual = str.charAt(i);
            String restante = str.substring(0, i) + str.substring(i + 1);

            permutar(restante, resultado + actual);
        }
    }

    public static void main(String[] args) {
        permutar("ABC", "");
    }
}

public class Main {

    static HashMap<Integer, Integer> memo = new HashMap<>();

    public static int fibonacci(int n) {

        if (n <= 1) return n;

        if (memo.containsKey(n))
            return memo.get(n);

        int result = fibonacci(n - 1) + fibonacci(n - 2);
        memo.put(n, result);

        return result;
    }

    public static void main(String[] args) {
        System.out.println(fibonacci(40));
    }
}

public class Main {

    public static int fibonacci(int n) {

        if (n <= 1) return n;

        int[] dp = new int[n + 1];

        dp[0] = 0;
        dp[1] = 1;

        for (int i = 2; i <= n; i++)
            dp[i] = dp[i - 1] + dp[i - 2];

        return dp[n];
    }

    public static void main(String[] args) {
        System.out.println(fibonacci(40));
    }
}

-> two pointers ---
encontrar dos números que sumen 
target en array ordenado.

public class Main {

    public static void twoSum(int[] arr, int target) {

        int left = 0;
        int right = arr.length - 1;

        while (left < right) {

            int sum = arr[left] + arr[right];

            if (sum == target) {
                System.out.println(arr[left] + " " + arr[right]);
                return;
            }

            if (sum < target)
                left++;
            else
                right--;
        }
    }

    public static void main(String[] args) {
        int[] arr = {1,2,3,4,6,8};
        twoSum(arr, 10);
    }
}

-> sliding window ---
Subarray con suma máxima de tamaño k

public class Main {

    public static int maxSum(int[] arr, int k) {

        int windowSum = 0;

        for (int i = 0; i < k; i++)
            windowSum += arr[i];

        int maxSum = windowSum;

        for (int i = k; i < arr.length; i++) {

            windowSum += arr[i] - arr[i - k];
            maxSum = Math.max(maxSum, windowSum);
        }

        return maxSum;
    }

    public static void main(String[] args) {

        int[] arr = {2,1,5,1,3,2};
        System.out.println(maxSum(arr, 3));
    }
}

-> observer ---
Notifica cambios automáticamente.

import java.util.ArrayList;
import java.util.List;

interface Observer {
    void actualizar(String mensaje);
}

class Usuario implements Observer {
    private String nombre;

    public Usuario(String nombre) {
        this.nombre = nombre;
    }

    public void actualizar(String mensaje) {
        System.out.println(nombre + " recibió: " + mensaje);
    }
}

class Canal {
    private List<Observer> observadores = new ArrayList<>();

    public void suscribir(Observer o) {
        observadores.add(o);
    }

    public void notificar(String mensaje) {
        for (Observer o : observadores) {
            o.actualizar(mensaje);
        }
    }
}

public class Main {
    public static void main(String[] args) {
        Canal canal = new Canal();
        canal.suscribir(new Usuario("Carlos"));
        canal.notificar("Nuevo video!");
    }
}

Presentation
Domain
Data
Y se utiliza un patrón de arquitectura como:
MVC (antiguo)
MVP
MVVM (el estándar actual)

-> MVVM ---
Model → Datos
View → UI (Activity / Fragment)
ViewModel → Lógica de presentación
View → ViewModel → UseCase → Repository → DataSource

-> clean architecture ---
Divide el proyecto en capas independientes.
Capas: Las capas externas dependen de las internas, NUNCA al revés.
Presentation
Domain
Data

estructura profesional
com.miapp
 ├── data
 │    ├── repository
 │    └── datasource
 ├── domain
 │    ├── model
 │    ├── repository
 │    └── usecase
 └── presentation
      ├── view
      └── viewmodel
